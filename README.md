## üéØ K-Batch System Deep Dive

### **Batch Calculation Logic**
```javascript
// Current year: 2024
// K24 = Admitted in 2024 (1st Year)
// K23 = Admitted in 2023 (2nd Year)
// K22 = Admitted in 2022 (3rd Year)
// K21 = Admitted in 2021 (4th Year)

const getKBatch = (admissionYear) => {
  return `K${admissionYear.toString().slice(-2)}`;
};

const getCurrentYear = (kBatch) => {
  const admissionYear = parseInt(`20${kBatch.slice(1)}`);
  const currentYear = new Date().getFullYear();
  return currentYear - admissionYear + 1;
};
```

// Hashtag parsing function
const parseHashtags = (text) => {
  const hashtagRegex = /#[a-zA-Z0-9_]+/g;
  return text.replace(hashtagRegex, (match) => 
    `<a href="/hashtags/${match.slice(1)}" class="hashtag-link">${match}</a>`
  );
};
